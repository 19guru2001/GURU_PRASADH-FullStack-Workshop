let count = 0
let step = 1
const counter = document.getElementById('counter')
const inc = document.getElementById('increment')
const dec = document.getElementById('decrement')
const reset = document.getElementById('reset')
const steps = document.querySelectorAll('.step-btn')

function render() {
    counter.textContent = count
    counter.style.color = count>0?'green':count<0?'red':'black'
}

inc.addEventListener('click', () => { count += step; render() })
dec.addEventListener('click', () => { count = Math.max(0,count-step); render() })
reset.addEventListener('click', () => { count = 0; render() })
steps.forEach(b=>b.addEventListener('click', ()=>{ step=+b.dataset.step }))

render()

form-validator.js
const username = document.getElementById('username')
const email = document.getElementById('email')
const password = document.getElementById('password')
const confirmPass = document.getElementById('confirm')
const submitBtn = document.getElementById('submitBtn')

const usernameError = document.getElementById('usernameError')
const emailError = document.getElementById('emailError')
const passwordError = document.getElementById('passwordError')
const confirmError = document.getElementById('confirmError')

function validateUsername() {
    const val = username.value
    if(!/^[a-zA-Z0-9]{3,15}$/.test(val)){ usernameError.textContent='Invalid'; return false }
    usernameError.textContent='✓'; return true
}

function validateEmail() {
    const val = email.value
    if(!/^\S+@\S+\.\S+$/.test(val)){ emailError.textContent='Invalid'; return false }
    emailError.textContent='✓'; return true
}

function validatePassword() {
    const val = password.value
    const regex = /^(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#$%^&*()_+\-=\[\]{};':"\\|,.<>\/?]).{8,}$/
    if(!regex.test(val)){ passwordError.textContent='Invalid'; return false }
    passwordError.textContent='✓'; return true
}

function validateConfirm() {
    if(confirmPass.value !== password.value){ confirmError.textContent='Mismatch'; return false }
    confirmError.textContent='✓'; return true
}

function checkForm() {
    submitBtn.disabled = !(validateUsername() && validateEmail() && validatePassword() && validateConfirm())
}

username.addEventListener('blur', ()=>{ validateUsername(); checkForm() })
email.addEventListener('blur', ()=>{ validateEmail(); checkForm() })
password.addEventListener('blur', ()=>{ validatePassword(); checkForm() })
confirmPass.addEventListener('blur', ()=>{ validateConfirm(); checkForm() })

document.getElementById('regForm').addEventListener('submit', e=>e.preventDefault())

todo.js
const form = document.getElementById('todoForm')
const input = document.getElementById('todoInput')
const categorySelect = document.getElementById('category')
const list = document.getElementById('todoList')
const filters = document.querySelectorAll('#filters button')
const counts = document.getElementById('counts')
let todos = JSON.parse(localStorage.getItem('todos'))||[]
function save(){ localStorage.setItem('todos',JSON.stringify(todos)) }
function render(filter='All'){
list.innerHTML=''
let filtered=todos.filter(t=>filter==='All'||t.category===filter)
filtered.forEach((t,i)=>{
const li=document.createElement('li')
const text=document.createElement('span')
text.textContent=t.text+' ['+t.category+']'
if(t.completed) text.classList.add('completed')
text.addEventListener('click',()=>{ t.completed=!t.completed; save(); render(filter) })
const del=document.createElement('span')
del.textContent='X'
del.style.color='red'
del.addEventListener('click',()=>{ todos.splice(i,1); save(); render(filter) })
li.appendChild(text)
li.appendChild(del)
list.appendChild(li)
})
const workCount=todos.filter(t=>t.category==='Work').length
const personalCount=todos.filter(t=>t.category==='Personal').length
counts.textContent=`Work: ${workCount}, Personal: ${personalCount}, Total: ${todos.length}`
}
form.addEventListener('submit',e=>{ e.preventDefault(); if(input.value.trim()==='') return; todos.push({text:input.value,category:categorySelect.value,completed:false}); input.value=''; save(); render() })
filters.forEach(btn=>btn.addEventListener('click',()=>render(btn.dataset.filter)))
render()

{ title:'Overview', content:'Overview content here...' },
{ title:'Features', content:'Features content here...' },
{ title:'Pricing', content:'Pricing content here...' }
]
const tabsContainer=document.getElementById('tabs')
const contentContainer=document.getElementById('content')
let activeIndex=0
function renderTabs(){
tabsContainer.innerHTML=''
tabsData.forEach((tab,i)=>{
const div=document.createElement('div')
div.textContent=tab.title
div.className='tab'+(i===activeIndex?' active':'')
div.addEventListener('click',()=>{ activeIndex=i; renderTabs() })

})
contentContainer.textContent=tabsData[activeIndex].content
}

function createModal({title,content,buttons}){
const overlay=document.getElementById('overlay')
const modal=document.createElement('div')
modal.className='modal closed'
modal.innerHTML=`<h2>${title}</h2><div>${content}</div>`
buttons.forEach(b=>{
const btn=document.createElement('button')
btn.textContent=b.text
btn.addEventListener('click',()=>{ b.onClick(modal,overlay) })
btn.style.background=b.type==='danger'?'red':b.type==='secondary'?'gray':'blue'
btn.style.color='white'
modal.appendChild(btn)
})
overlay.appendChild(modal)
overlay.addEventListener('click',e=>{ if(e.target===overlay) closeModal(modal,overlay) })
document.addEventListener('keydown',e=>{ if(e.key==='Escape') closeModal(modal,overlay) })
return { open: ()=>{ overlay.style.display='flex'; modal.className='modal open' }, close: ()=>closeModal(modal,overlay) }
}
function closeModal(modal,overlay){ modal.className='modal closed'; setTimeout(()=>overlay.style.display='none',300) }
document.getElementById('openModal').addEventListener('click',()=>{
const modal=createModal({title:'Confirm Delete',content:'Are you sure?',buttons:[{text:'Cancel',type:'secondary',onClick:closeModal},{text:'Delete',type:'danger',onClick:(m,o)=>{ console.log('Deleted'); closeModal(m,o) }}]})
modal.open()
})
