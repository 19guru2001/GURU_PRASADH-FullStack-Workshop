const container=document.getElementById('accordion')
accordionData.forEach(d=>{
const section=document.createElement('div'); section.className='section'
const header=document.createElement('div'); header.className='header'
header.innerHTML=`<span>${d.title}</span><span class="arrow">â–¶</span>`
const content=document.createElement('div'); content.className='content'; content.textContent=d.content
header.addEventListener('click',()=>{ const open=header.classList.toggle('open'); content.style.maxHeight=open?content.scrollHeight+'px':'0' })

})
