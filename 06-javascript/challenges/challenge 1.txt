let count = 0
let step = 1
const counter = document.getElementById('counter')
const inc = document.getElementById('increment')
const dec = document.getElementById('decrement')
const reset = document.getElementById('reset')
const steps = document.querySelectorAll('.step-btn')

function render() {
    counter.textContent = count
    counter.style.color = count>0?'green':count<0?'red':'black'
}

inc.addEventListener('click', () => { count += step; render() })
dec.addEventListener('click', () => { count = Math.max(0,count-step); render() })
reset.addEventListener('click', () => { count = 0; render() })
steps.forEach(b=>b.addEventListener('click', ()=>{ step=+b.dataset.step }))

render()
